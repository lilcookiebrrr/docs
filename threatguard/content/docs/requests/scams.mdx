### JavaScript Examples


#### Text Classification
```js
const fetch = require("node-fetch");

(async () => {
  const response = await fetch("https://scams.rip/api/scan", {
    method: "POST",
    headers: {
      "apikey": "XXXXX-XXXXX-XXXXX-XXXXX-XXXXX",
      "Content-Type": "application/json"
    },
    body: JSON.stringify({ 
        category: "scam",
        type: "text",
        content: "Hi, how are you friend.",
    })
  });

  const data = await response.json();
  console.log(data);
})();
```
